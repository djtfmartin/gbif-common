016package org.gbif.utils.file;
017
018import org.gbif.utils.file.properties.PropertiesUtil;
019
020import java.util.Properties;
021
022import org.junit.Test;
023
024import static junit.framework.Assert.assertEquals;
025import static org.junit.Assert.assertNotNull;
026import static org.junit.Assert.assertTrue;
027
028/**
029 * Tests related to {@link PropertiesUtil}
030 */
031public class PropertiesUtilTest {
032
033  @Test
034  public void testFilterProperties(){
035
036    Properties properties = new Properties();
037    properties.put("prefix.key1", "value1");
038    properties.put("key2", "value2");
039
040    Properties filteredProperties = PropertiesUtil.filterProperties(properties, "prefix.");
041    assertEquals(1, filteredProperties.size());
042    assertTrue("Prefix is removed from the original key", filteredProperties.containsKey("key1"));
043    assertEquals("Value remains the same", properties.get("prefix.key1"), filteredProperties.getProperty("key1"));
044  }
045
046  @Test
047  public void testSubsetProperties(){
048
049    Properties properties = new Properties();
050    properties.put("prefix.key1", "value1");
051    properties.put("key2", "value2");
052
053    Properties newProperties = PropertiesUtil.subsetProperties(properties, "prefix.");
054    assertEquals(1, newProperties.size());
055    assertEquals(2, properties.size());
056
057    assertTrue("Prefix is kept from the original key", newProperties.containsKey("prefix.key1"));
058    assertTrue("key1 is still in original Properties", properties.containsKey("prefix.key1"));
059    assertTrue("key2 is still in original Properties", properties.containsKey("key2"));
060  }
061
062  @Test
063  public void testRemoveProperties(){
064
065    Properties properties = new Properties();
066    properties.put("prefix.key1", "value1");
067    properties.put("key2", "value2");
068
069    Properties newProperties = PropertiesUtil.removeProperties(properties, "prefix.");
070    assertEquals(1, newProperties.size());
071    assertEquals(1, properties.size());
072
073    assertTrue("Prefix is kept from the original key", newProperties.containsKey("prefix.key1"));
074    assertTrue("Other element is still present in original Propeties", properties.containsKey("key2"));
075  }
076
077  @Test
078  public void testEmptyProperties(){
079    Properties properties = new Properties();
080    Properties newProperties = PropertiesUtil.removeProperties(properties, "prefix");
081    assertNotNull(newProperties);
082    assertEquals(0, properties.size());
083  }
084
085  @Test(expected = NullPointerException.class)
086  public void testExceptionNoProperties(){
087    PropertiesUtil.removeProperties(null, "prefix");
088  }
089
090  @Test(expected = IllegalStateException.class)
091  public void testExceptionNoPrefix(){
092    Properties properties = new Properties();
093    properties.put("prefix.key1", "value1");
094    PropertiesUtil.removeProperties(properties, null);
095  }
096}
