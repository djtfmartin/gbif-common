001package org.gbif.utils.file.properties;
002
003import java.io.File;
004import java.io.FileInputStream;
005import java.io.FileReader;
006import java.io.IOException;
007import java.io.UnsupportedEncodingException;
008import java.net.URL;
009import java.util.Properties;
010
011import com.google.common.base.Strings;
012import com.google.common.io.Closer;
013import com.google.common.io.Resources;
014
015/**
016 * Utility class for handling properties files.
017 */
018public class PropertiesUtil {
019
020  /**
021   * When we encode strings, we always specify UTF8 encoding
022   */
023  public static final String UTF8_ENCODING = "UTF-8";
024
025  /**
026   * Private default constructor.
027   */
028  private PropertiesUtil() {
029    // empty block
030  }
031
032  /**
033   * Loads a properties file.
034   * The file should be available in the classpath, the default {@link ClassLoader} is used to load the file.
035   * 
036   * @throws IOException Should there be an issue in loading the file
037   * @throws IllegalArgumentException If the file does not exist
038   */
039  public static Properties loadProperties(String propertiesFile) throws IOException, IllegalArgumentException {
040    Properties tempProperties = new Properties();
041    Closer closer = Closer.create();
042    try {
043      File file = new File(propertiesFile);
044      if (file.exists()) {// first tries to load the file as a external file
045        tempProperties.load(closer.register(new FileInputStream(file)));
046      } else { // tries to load the file as a resource
047        URL configFileURL = Resources.getResource(propertiesFile);
048        tempProperties.load(closer.register(Resources.asByteSource(configFileURL).openStream()));
049      }
050    } finally {
051      closer.close();
052    }
053    return tempProperties;
054  }
055
056  /**
057   * Reads a property file from an absolute filepath.
058   */
059  public static Properties readFromFile(String filepath) throws IOException, IllegalArgumentException {
060    if (Strings.isNullOrEmpty(filepath)) {
061      throw new IllegalArgumentException("No properties file given");
062    }
063    File pf = new File(filepath);
064    if (!pf.exists()) {
065      throw new IllegalArgumentException("Cannot find properties file " + filepath);
066    }
067    Properties properties = new Properties();
068
069    Closer closer = Closer.create();
070    try {
071      FileReader reader = closer.register(new FileReader(pf));
072      properties.load(reader);
073    } finally {
074      closer.close();
075    }
076    return properties;
077  }
078
079  /**
080   * Reads and casts the named property as an Double.
081   * 
082   * @param p The properties file to read from.
083   * @param key To read the value of.
084   * @param exceptionForNull If true, and the property is not found an IAE is thrown, otherwise defaultValue is
085   *        returned
086   * @param defaultValue If the property is not found, and exceptionForNull is false, this is returned for missing
087   *        properties.
088   * @return The property at the key as an Double
089   * @throws IllegalArgumentException if the property is invalid (can't be cast to a double) or not found and we are
090   *         instructed to throw it.
091   */
092  public static Double propertyAsDouble(Properties p, String key, boolean exceptionForNull, Double defaultValue)
093    throws IllegalArgumentException {
094    String v = p.getProperty(key);
095    if (v != null) {
096      try {
097        return Double.parseDouble(v);
098      } catch (NumberFormatException e) {
099        throw new IllegalArgumentException("Invalid value[" + v + "] supplied for " + key);
100      }
101    } else {
102      if (exceptionForNull) {
103        throw new IllegalArgumentException("Missing property for " + key);
104      } else {
105        return defaultValue;
106      }
107    }
108  }
109

172  /**
173   * Reads and converts the named property as UTF8 bytes.
174   * 
175   * @param p The properties file to read from.
176   * @param key To read the value of.
177   * @param exceptionForNull If true, and the property is not found an IAE is thrown, otherwise defaultValue is
178   *        returned
179   * @param defaultValue If the property is not found, and exceptionForNull is false, this is returned for missing
180   *        properties.
181   * @return The property at the key as byte[]t
182   * @throws IllegalArgumentException if the property is not found and we are instructed to throw it.
183   */
184  public static byte[] propertyAsUTF8Bytes(Properties p, String key, boolean exceptionForNull, byte[] defaultValue)
185    throws IllegalArgumentException {
186    String v = p.getProperty(key);
187    if (v != null) {
188      try {
189        return v.getBytes(UTF8_ENCODING);
190      } catch (UnsupportedEncodingException e) {
191        // never one would hope
192        throw new RuntimeException("System does not support " + UTF8_ENCODING + " encoding");
193      }
194    } else {
195      if (exceptionForNull) {
196        throw new IllegalArgumentException("Missing property for " + key);
197      } else {
198        return defaultValue;
199      }
200    }
201  }
