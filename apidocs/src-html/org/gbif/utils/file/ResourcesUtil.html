001package org.gbif.utils.file;
002
003import java.io.File;
004import java.io.FileOutputStream;
005import java.io.IOException;
006import java.io.OutputStream;
007import java.net.URL;
008
009import com.google.common.io.Closer;
010import com.google.common.io.Files;
011import com.google.common.io.Resources;
012import org.slf4j.Logger;
013import org.slf4j.LoggerFactory;
014
015/**
016 * Utils class dealing with classpath resources in addition to guavas {@link Resources}.
017 */
018public class ResourcesUtil {
019
020  private static final Logger LOG = LoggerFactory.getLogger(ResourcesUtil.class);
021
022  /**
023   * Static utils class.
024   */
025  private ResourcesUtil() {
026
027  }
028
029  /**
030   * Copies classpath resources to real files.
031   *
032   * @param folder                 to copy resource files into
033   * @param ignoreMissingResources if true ignores missing resources, throws IOException otherwise
034   * @param classpathPrefix        common prefix added to all classpath resources which is not used for the result file
035   *                               path.
036   * @param classpathResources     list of classpath resources to be copied into folder
037   */
038  public static void copy(File folder, String classpathPrefix, boolean ignoreMissingResources,
039    String... classpathResources) throws IOException {
040    for (String classpathResource : classpathResources) {
041      String res = classpathPrefix + classpathResource;
042      URL url = null;
043      try {
044        url = Resources.getResource(res);
045        if (url == null) {
046          throw new IllegalArgumentException("Classpath resource " + res + " not existing");
047        }
048      } catch (IllegalArgumentException e) {
049        if (ignoreMissingResources) {
050          LOG.debug("Resource {} not found", res);
051          continue;
052        }
053        throw new IOException(e);
054      }
055
056      Closer closer = Closer.create();
057      try {
058        File f = new File(folder, classpathResource);
059        Files.createParentDirs(f);
060        OutputStream out = closer.register(new FileOutputStream(f));
061        Resources.copy(url, out);
062      } catch (Throwable e) {
063        // must catch Throwable for closer to work, see https://code.google.com/p/guava-libraries/wiki/ClosingResourcesExplained
064        throw closer.rethrow(e);
065      } finally {
066        closer.close();
067      }
068    }
069  }
070
071}
