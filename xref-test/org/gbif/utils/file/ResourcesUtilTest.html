16    @Test
17    public void testCopyResources() throws Exception {
18      File tmp = FileUtils.createTempDir();
19      tmp.deleteOnExit();
20      ResourcesUtil.copy(tmp, "", true, "testNOT_EXISTING.txt", "test1/test.txt", "test1/test2/test.txt");
21  
22      // test
23      File t1 = new File(tmp, "test1/test.txt");
24      File t2 = new File(tmp, "test1/test2/test.txt");
25      org.apache.commons.io.FileUtils.contentEquals(t1, t2);
26      assertTestFile(t1);
27      assertTestFile(t2);
28    }
29  
30    @Test(expected = IOException.class)
31    public void testCopyResourcesThrowing() throws Exception {
32      File tmp = FileUtils.createTempDir();
33      tmp.deleteOnExit();
34      ResourcesUtil.copy(tmp, "", false, "testNOT_EXISTING.txt", "test1/test.txt", "test1/test2/test.txt");
35  
36      // test
37      File t1 = new File(tmp, "test1/test.txt");
38      File t2 = new File(tmp, "test1/test2/test.txt");
39      org.apache.commons.io.FileUtils.contentEquals(t1, t2);
40      assertTestFile(t1);
41      assertTestFile(t2);
42    }
43  
44    @Test
45    public void testList() throws Exception {
46      assertEquals(Sets.newHashSet("test.txt", "test2"), Sets.newHashSet(ResourcesUtil.list(ResourcesUtil.class, "test1")));
47      assertEquals(Sets.newHashSet("test.txt"), Sets.newHashSet(ResourcesUtil.list(ResourcesUtil.class, "test1/test2")));
48      assertEquals(Sets.newHashSet("utf16be.xml", "utf16le.xml", "utf8.xml", "utf8bom.xml"), Sets.newHashSet(ResourcesUtil.list(ResourcesUtil.class, "sax")));
49      assertEquals(Sets.newHashSet(), Sets.newHashSet(ResourcesUtil.list(ResourcesUtil.class, "abba")));
50    }
51  
52    private void assertTestFile(File tf) throws FileNotFoundException {
53      assertEquals("hallo\n", isu.readEntireStream(FileUtils.getInputStream(tf)));
54    }
