17    /**
18     * This test is simply to illustrate why we use the BigDecimal(String) constructor.
19     * new BigDecimal(23.1) equals 23.10000000000000142108547152020037174224853515625
20     */
21    @Test
22    public void testBigDecimalConstructor(){
23      assertFalse(new BigDecimal(23.1d).equals(new BigDecimal("23.1")));
24    }
25  
26    @Test(expected = NullPointerException.class)
27    public void testBigDecimalConstructorNull(){
28      BigDecimalUtils.fromDouble(null, false);
29    }

31    /**
32     * This test is simply to show why we are not using BigDecimal stripTrailingZeros function.
33     */
34    @Test
35    public void demonstrateBigDecimalStripTrailingZerosEquality() {
36      assertFalse(new BigDecimal("500").equals(new BigDecimal("500.0").stripTrailingZeros()));
37  
38      // but, compareTo function will work as expected
39      assertEquals(0, new BigDecimal("500").compareTo(new BigDecimal("500.0").stripTrailingZeros()));
40    }

42    @Test
43    public void testFromDouble(){
44      assertEquals(new BigDecimal("23.4"), BigDecimalUtils.fromDouble(23.4, true));
45      assertEquals(new BigDecimal("23.4"), BigDecimalUtils.fromDouble(23.4, false));
46  
47      assertEquals(new BigDecimal("23"), BigDecimalUtils.fromDouble(23d, true));
48      assertEquals(new BigDecimal("23.0"), BigDecimalUtils.fromDouble(23d, false));
49  
50      assertEquals(new BigDecimal("23"), BigDecimalUtils.fromDouble(23.0, true));
51      assertEquals(new BigDecimal("23.0"), BigDecimalUtils.fromDouble(23.0, false));
52  
53      assertEquals(new BigDecimal("23"), BigDecimalUtils.fromDouble(23.0000000000, true));
54      assertEquals(new BigDecimal("23.0"), BigDecimalUtils.fromDouble(23.0000000000, false));
55  
56      assertEquals(new BigDecimal("23.000000000001"), BigDecimalUtils.fromDouble(23.000000000001, true));
57      assertEquals(new BigDecimal("23.000000000001"), BigDecimalUtils.fromDouble(23.000000000001, false));
58  
59      assertEquals(new BigDecimal("500"), BigDecimalUtils.fromDouble(500d, true));
60      assertEquals(new BigDecimal("500.0"), BigDecimalUtils.fromDouble(500d, false));
61  
62      assertEquals(new BigDecimal("50000"), BigDecimalUtils.fromDouble(50000d, true));
63      assertEquals(new BigDecimal("50000.0"), BigDecimalUtils.fromDouble(50000d, false));
64  
65      assertEquals(new BigDecimal("50000.01"), BigDecimalUtils.fromDouble(50000.01, true));
66      assertEquals(new BigDecimal("50000.01"), BigDecimalUtils.fromDouble(50000.01, false));
67    }
