15    @Test
16    public void testLineComparatorCSV() {
17      LineComparator comp = new LineComparator(3, ",", '"');
18      List<String> lines = new ArrayList<String>();
19      String l1 = "121,432423,9099053,\"Frieda karla L.,Ahrens\"";
20      String l2 = "adshjhg,fsdfsd,fdsfdsfsd,Forunkel forunculus,Janssen";
21      String l3 = ",,,,zzz";
22      String l4 = "321,2453,432423,berndef,ahrene32";
23      lines.add(l1);
24      lines.add(l2);
25      lines.add(l3);
26      lines.add(l4);
27  
28      Collections.sort(lines, comp);
29  
30      assertEquals(l2, lines.get(0));
31      assertEquals(l1, lines.get(1));
32      assertEquals(l4, lines.get(2));
33      assertEquals(l3, lines.get(3));
34    }

88     * Test for respecting equals and compareTo to ensure comparator respects java contracts.
89     * http://dev.gbif.org/issues/browse/POR-2730
90     */
91    @Test
92    public void testEqualsCompareToContract() throws IOException {
93  
94      LineComparator comp = new LineComparator(0, ",", null, IGNORE_CASE_COMPARATOR);
95  
96      String s1 = ",2,3"; // null in first instance
97      String s2 = ",2,3"; // null in first instance
98  
99      // http://docs.oracle.com/javase/7/docs/api/java/util/Comparator.html#compare(T,%20T)
100     int sign1 = (int) Math.signum(comp.compare(s1, s2));
101     int sign2 = (int) Math.signum(comp.compare(s2, s1));
102 
103     assertEquals(sign1 * -1, sign2);
