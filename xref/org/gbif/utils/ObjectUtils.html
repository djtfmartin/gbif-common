12    /**
13     * Returns the first of the given parameters that is not null.
14     * If all given parameters are null, returns null.
15     *
16     * @param items
17     * @param <T>
18     * @return
19     */
20    public static <T> T coalesce(T ... items) {
21      if (items != null) {
22        for (T i : items) if (i != null) return i;
23      }
24      return null;
25    }
