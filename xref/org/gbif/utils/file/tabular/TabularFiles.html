11  /**
12   * Static utility methods related to {@link TabularDataFileReader} instances.
13   *
14   */
15  public class TabularFiles {
16  
17  
18    /**
19     * Get a new TabularDataFileReader.
20     *
21     * @param in
22     * @param quoteChar
23     * @param delimiterChar
24     * @param endOfLineSymbols
25     * @param charset
26     * @param headerLine
27     * @return
28     */
29    public static TabularDataFileReader<List<String>> newTabularFileReader(InputStream in, char quoteChar, char delimiterChar,
30                                                             String endOfLineSymbols, Charset charset, boolean headerLine){
31  
32      Preconditions.checkNotNull(in, "An InputStream must be provided");
33      Preconditions.checkNotNull(charset, "A Charset must be provided");
34      return new SuperCsvFileReader(in, quoteChar, delimiterChar, endOfLineSymbols, charset, headerLine);
35    }
36  
37  
38    /**
39     * Get a new TabularDataFileReader using UTF-8 charset, default quote char and default endOfLineSymbols.
40     *
41     * @param in
42     * @param delimiterChar
43     * @param headerLine
44     */
45    public static TabularDataFileReader<List<String>> newTabularFileReader(InputStream in, char delimiterChar,
46                                                                           boolean headerLine){
47      return newTabularFileReader(in, CsvPreference.STANDARD_PREFERENCE.getQuoteChar(),
48              delimiterChar, CsvPreference.STANDARD_PREFERENCE.getEndOfLineSymbols(), Charsets.UTF_8, headerLine);
49    }

51  }
