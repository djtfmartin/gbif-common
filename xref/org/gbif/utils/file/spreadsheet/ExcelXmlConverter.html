52  /**
53   * An XLSX → CSV processor, which is a copy of <a href="https://svn.apache.org/repos/asf/poi/trunk/src/examples/src/org/apache/poi/xssf/eventusermodel/XLSX2CSV.java">ExcelXmlConverter.java</a>
54   * modified to choose a particular sheet and output to a {@link Writer}. Dates are formatted to ISO 8601 standard,
55   * otherwise the formats defined in Excel are used.
56   * <p>
57   * The SAX parser should keep memory usage low, so large workbooks can be handled.
58   * <p>
59   * Original JavaDoc follows:
60   * <p>
61   * A rudimentary XLSX -> CSV processor modeled on the
62   * POI sample program XLS2CSVmra from the package
63   * org.apache.poi.hssf.eventusermodel.examples.
64   * As with the HSSF version, this tries to spot missing
65   *  rows and cells, and output empty entries for them.
66   * <p>
67   * Data sheets are read using a SAX parser to keep the
68   * memory footprint relatively small, so this should be
69   * able to read enormous workbooks.  The styles table and
70   * the shared-string table must be kept in memory.  The
71   * standard POI styles table class is used, but a custom
72   * (read-only) class is used for the shared string table
73   * because the standard POI SharedStringsTable grows very
74   * quickly with the number of unique strings.
75   * <p>
76   * For a more advanced implementation of SAX event parsing
77   * of XLSX files, see {@link XSSFEventBasedExcelExtractor}
78   * and {@link XSSFSheetXMLHandler}. Note that for many cases,
79   * it may be possible to simply use those with a custom
80   * {@link SheetContentsHandler} and no SAX code needed of
81   * your own!
82   */
83  public class ExcelXmlConverter {
84    private static final Logger LOG = LoggerFactory.getLogger(ExcelXmlConverter.class);
85  
86    /**
87     * Convert a workbook (.xlsx format) to some other format like CSV.
88     *
89     * Example: ExcelXmlConverter.convert(xlsxPath, new CsvSpreadsheetConsumer(new FileWriter(outputCsvFile)));
90     *
91     * @param workbookFile Path to the workbook file
92     * @param writer A writer which will be given many List<String>s
93     * @throws IOException
94     * @throws InvalidFormatException Thrown if invalid XML is found whilst parsing an input SpreadsheetML file.
95     */
96    public static long convert(Path workbookFile, SpreadsheetConsumer writer) throws IOException, SAXException, OpenXML4JException {
97      OPCPackage p = OPCPackage.open(workbookFile.toFile(), PackageAccess.READ);
98      ExcelXmlConverter converter = new ExcelXmlConverter(p, writer);
99  
100     long count = converter.process();
101 
102     writer.flush();
103     return count;
104   }

106   /**
107    * Uses the XSSF Event SAX helpers to do most of the work of parsing the Sheet XML, and output the contents
108    * to a SpreadsheetConsumer.
109    */
110   private class SheetToCSV implements SheetContentsHandler {
111     private int currentRow = -1;
112     private int currentCol = -1;
113 
114     /**
115      * Number of columns to read starting with leftmost
116      */
117     private final long numColumns;
118 
119     /**
120      * Destination for data
121      */
122     private final SpreadsheetConsumer writer;
123 
124     private List<String> row = new ArrayList<>();
125 
126     public SheetToCSV(SpreadsheetConsumer writer, long minColumns) {
127       this.writer = writer;
128       this.numColumns = minColumns;
129     }
130 
131     private void outputMissingRows(int number) {
132       for (int i = 0; i < number; i++) {
133         for (int j = 0; j < numColumns; j++) {
134           row.add("");
135         }
136         try {
137           writer.write(row);
138         } catch (IOException e) {
139           e.printStackTrace();
140         }
141       }
142     }
143 
144     @Override
145     public void startRow(int rowNum) {
146       // If there were gaps, output the missing rows
147       outputMissingRows(rowNum - currentRow - 1);
148       // Prepare for this row
149       currentRow = rowNum;
150       currentCol = -1;
151     }
152 
153     @Override
154     public void endRow(int rowNum) {
155       // Ensure the correct number of columns
156       for (int i = currentCol+1; i < numColumns; i++) {
157         row.add("");
158       }
159       try {
160         writer.write(row);
161         row = new ArrayList<>(row.size());
162       } catch (IOException e) {
163         LOG.error("Error reading spreadsheet " + e.getMessage(), e);
164       }
165     }
166 
167     @Override
168     public void cell(String cellReference, String formattedValue, XSSFComment comment) {
169       // gracefully handle missing CellRef here in a similar way as XSSFCell does
170       if (cellReference == null) {
171         cellReference = new CellAddress(currentRow, currentCol).formatAsString();
172       }
173 
174       int thisCol = (new CellReference(cellReference)).getCol();
175 
176       if (thisCol < numColumns) {
177         // Did we miss any cells?
178         int missedCols = thisCol - currentCol - 1;
179         for (int i = 0; i < missedCols; i++) {
180           row.add("");
181         }
182         currentCol = thisCol;
183 
184         row.add(formattedValue);
185       } else {
186         LOG.debug("Not including cell {} “{}”; it is outside the main table", cellReference, formattedValue);
187       }
188     }
189 
190     @Override
191     public void headerFooter(String text, boolean isHeader, String tagName) {}
192   }
193 
194   /**
195    * Counts the number of cells with data in the sheet.
196    */
197   private class CountSheetData implements SheetContentsHandler {
198 
199     final Long[] cellCount;
200     final Long[] headerLength;
201 
202     CountSheetData(Long[] cellCount, Long[] headerLength) {
203       this.cellCount = cellCount;
204       this.headerLength = headerLength;
205       cellCount[0] = 0L;
206       headerLength[0] = 0L;
207     }
208 
209     @Override
210     public void startRow(int rowNum) {}
211 
212     @Override
213     public void endRow(int rowNum) {
214       if (rowNum == 0) {
215         headerLength[0] = cellCount[0];
216       }
217     }
218 
219     @Override
220     public void cell(String cellReference, String formattedValue, XSSFComment comment) {
221       if (formattedValue != null) {
222         cellCount[0]++;
223       }
224     }
225 
226     @Override
227     public void headerFooter(String text, boolean isHeader, String tagName) {}
228   }

235   /**
236    * Creates a new XLSX -> CSV examples
237    *
238    * @param pkg    The XLSX package to process
239    * @param writer The writer to output to
240    */
241   public ExcelXmlConverter(OPCPackage pkg, SpreadsheetConsumer writer) {
242     this.xlsxPackage = pkg;
243     this.writer = writer;
244   }

246   /**
247    * Parses and shows the content of one sheet
248    * using the specified styles and shared-strings tables.
249    *
250    * @param styles The table of styles that may be referenced by cells in the sheet
251    * @param strings The table of strings that may be referenced by cells in the sheet
252    * @param sheetInputStream The stream to read the sheet-data from.
253 
254    * @exception java.io.IOException An IO exception from the parser,
255    *      possibly from a byte stream or character stream
256    *      supplied by the application.
257    * @throws SAXException if parsing the XML data fails.
258    */
259   public long processSheet(
260       StylesTable styles,
261       ReadOnlySharedStringsTable strings,
262       SheetContentsHandler sheetHandler,
263       InputStream sheetInputStream
264   ) throws IOException, SAXException {
265     IsoDateDataFormatter formatter = new IsoDateDataFormatter();
266     InputSource sheetSource = new InputSource(sheetInputStream);
267     XMLReader sheetParser = XMLReaderFactory.createXMLReader("org.apache.xerces.parsers.SAXParser");
268     ContentHandler handler = new XSSFSheetXMLHandler(styles, null, strings, sheetHandler, formatter, false);
269     sheetParser.setContentHandler(handler);
270     sheetParser.parse(sheetSource);
271     return writer.getCount();
272   }
273 
274   /**
275    * Initiates the processing of the XLS workbook file to CSV.
276    *
277    * @throws IOException If reading the data from the package fails.
278    * @throws SAXException if parsing the XML data fails.
279    */
280   public long process() throws IOException, OpenXML4JException, SAXException {
281     ReadOnlySharedStringsTable strings = new ReadOnlySharedStringsTable(this.xlsxPackage);
282 
283     // First find the sheet with the most data.
284     XSSFReader xssfReader = new XSSFReader(this.xlsxPackage);
285     StylesTable styles = xssfReader.getStylesTable();
286     XSSFReader.SheetIterator iter = (XSSFReader.SheetIterator) xssfReader.getSheetsData();
287 
288     long maxCount = -1;
289     int maxSheetIndex = -1;
290     long maxHeaderLength = -1;
291 
292     int index = 0;
293 
294     while (iter.hasNext()) {
295       try (InputStream stream = iter.next()) {
296         Long[] cellCount = new Long[1];
297         Long[] headerLength = new Long[1];
298 
299         String sheetName = iter.getSheetName();
300         LOG.debug("Counting cells in spreadsheet {} “{}”", index, sheetName);
301         processSheet(styles, strings, new CountSheetData(cellCount, headerLength), stream);
302 
303         if (cellCount[0] > maxCount) {
304           maxSheetIndex = index;
305           maxCount = cellCount[0];
306           maxHeaderLength = headerLength[0];
307         }
308       }
309       index++;
310     }
311 
312 
313     iter = (XSSFReader.SheetIterator) xssfReader.getSheetsData();
314     for (index = 0; index++ < maxSheetIndex; iter.next()) {}
315 
316     try (InputStream stream = iter.next()) {
317       String sheetName = iter.getSheetName();
318       LOG.info("Reading spreadsheet {} “{}” which has {} cells in {} columns", maxSheetIndex, sheetName, maxCount, maxHeaderLength);
319 
320       processSheet(styles, strings, new SheetToCSV(writer, maxHeaderLength), stream);
321     }
322 
323     // Then read and process that sheet.
324     return writer.getCount();
325   }
