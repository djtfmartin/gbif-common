7   import java.text.Format;
8   import java.text.SimpleDateFormat;
9   import java.util.Date;
10  import java.util.Locale;
11  
12  /**
13   * Override Excel formats to use ISO dates, but leave everything else as it was.
14   */
15  public class IsoDateDataFormatter extends DataFormatter {
16  
17    private DateFormat dateDateformat = new SimpleDateFormat("yyyy-MM-DD");
18    private DateFormat dateTimeDateformat = new SimpleDateFormat("yyyy-MM-DD'T'HH:mm:ss'Z'");
19  
20    public IsoDateDataFormatter() {
21      super(Locale.UK);
22    }
23  
24    private Format getDateFormat(String pFormatStr) {
25      // Within [] is an elapsed duration (hours?)
26  
27      // Just decide whether a time is present
28      if (pFormatStr.contains("h")) {
29        return dateTimeDateformat;
30      } else {
31        return dateDateformat;
32      }
33    }
34  
35    /**
36     * Formats a date to ISO 8601, otherwise hands over to the superclass.
37     */
38    public String formatRawCellContents(double value, int formatIndex, String formatString) {
39      // Is it a date?
40      if (DateUtil.isADateFormat(formatIndex, formatString) && DateUtil.isValidExcelDate(value)) {
41        Format dateFormat = getDateFormat(formatString);
42        Date d = DateUtil.getJavaDate(value, false);
43        return dateFormat.format(d);
44      }
45  
46      return super.formatRawCellContents(value, formatIndex, formatString, false);
47    }
48  }
