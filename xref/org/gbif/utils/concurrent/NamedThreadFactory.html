1   package org.gbif.utils.concurrent;
2   
3   import java.util.concurrent.Executors;
4   import java.util.concurrent.ThreadFactory;
5   import java.util.concurrent.atomic.AtomicInteger;
6   
7   /**
8    * Modified Executors DefaultThreadFactory to allow custom named thread pools.
9    * Otherwise, this factory yields the same semantics as the thread factory returned by
10   * {@link Executors#defaultThreadFactory()}.
11   */
12  public class NamedThreadFactory implements ThreadFactory {
13    private final ThreadGroup group;
14    private final AtomicInteger threadNumber = new AtomicInteger(1);
15    private final String namePrefix;
16  
17    /**
18     * @param poolName the name prefix of the thread pool which will be appended -number for the individual thread
19     */
20    public NamedThreadFactory(String poolName) {
21      SecurityManager s = System.getSecurityManager();
22      group = (s != null) ? s.getThreadGroup() :
23          Thread.currentThread().getThreadGroup();
24      namePrefix = poolName + "-";
25    }
26  
27    public Thread newThread(Runnable r) {
28      Thread t = new Thread(group, r,
29          namePrefix + threadNumber.getAndIncrement(),
30          0);
31      if (t.isDaemon())
32        t.setDaemon(false);
33      if (t.getPriority() != Thread.NORM_PRIORITY)
34        t.setPriority(Thread.NORM_PRIORITY);
35      return t;
36    }
37  }
