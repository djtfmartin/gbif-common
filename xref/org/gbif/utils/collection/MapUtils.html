6   import java.util.LinkedList;

8   import java.util.Map;

10  public class MapUtils {
11  
12    private MapUtils() {
13      throw new UnsupportedOperationException("Can't initialize class");
14    }
15  
16    /**
17     * This orders a Map by its values and returns a new {@link LinkedHashMap} which maintains that order.
18     *
19     * @param map to sort
20     * @param <K> type of the map key
21     * @param <V> type of the map value, this needs to implement {@link Comparable}
22     *
23     * @return a map ordered by the values of the input map
24     */
25    public static <K, V extends Comparable<? super V>> Map<K, V> sortByValue(Map<K, V> map) {
26      return sortByValueInternal(map, new Comparator<Map.Entry<K, V>>() {
27        public int compare(Map.Entry<K, V> o1, Map.Entry<K, V> o2) {
28          return o1.getValue().compareTo(o2.getValue());
29        }
30      });
31    }
32  
33    /**
34     * Order a Map by its values using the given value comparator and return a new {@link LinkedHashMap} which maintains that order.
35     *
36     * @param map to sort
37     * @param <K> type of the map key
38     * @param <V> type of the map value
39     *
40     * @return a map ordered by the values of the input map
41     */
42    public static <K, V> Map<K, V> sortByValue(Map<K, V> map, final Comparator<V> comparator) {
43      return sortByValueInternal(map, new Comparator<Map.Entry<K, V>>() {
44          public int compare(Map.Entry<K, V> o1, Map.Entry<K, V> o2) {
45            return comparator.compare(o1.getValue(), o2.getValue());
46          }
47        });
48    }
49  
50    private static <K, V> Map<K, V> sortByValueInternal(Map<K, V> map, final Comparator<Map.Entry<K, V>> comp) {
51      List<Map.Entry<K, V>> list = new LinkedList<Map.Entry<K, V>>(map.entrySet());
52      Collections.sort(list, comp);
53  
54      Map<K, V> result = new LinkedHashMap<K, V>(list.size());
55      for (Map.Entry<K, V> entry : list) {
56        result.put(entry.getKey(), entry.getValue());
57      }
58      return result;
59    }
60  
61    /**
62     * Returns a maps value or a given default if not present.
63     * For pre java8 code since j8 introduced Map.getOrDefault.
64     */
65    public static <K, V> V getOrDefault(Map<K,V> map, K key, V defaultValue) {
66      return map.containsKey(key) ? map.get(key) : defaultValue;
67    }
68  }
