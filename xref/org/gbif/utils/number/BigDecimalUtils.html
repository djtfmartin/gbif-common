7   /**
8    * Utility class to work with BigDecimal.
9    * See DoubleVsBigDecimal test class for more details about {@link BigDecimal}

11  public class BigDecimalUtils {
12  
13    private BigDecimalUtils(){}
14  
15    /**
16     * Convert a double to a BigDecimal.
17     *
18     * @param value non-null value to convert into a BigDecimal.
19     * @param stripTrailingZeros should the trailing zero(s) be removed? e.g. 25.0 would become 25
20     * @return instance of BigDecimal
21     */
22    public static BigDecimal fromDouble(Double value, boolean stripTrailingZeros){
23  
24      Preconditions.checkNotNull(value);
25  
26      //safer to create a BigDecimal from String than Double
27      BigDecimal bd = new BigDecimal(Double.toString(value));
28      if(stripTrailingZeros){
29        //we do not use stripTrailingZeros() simply because it plays with the scale and returns a BigDecimal
30        //that is numerically equals. see test in BigDecimalUtilsTest
31        if(bd.remainder(BigDecimal.ONE).movePointRight(bd.scale()).intValue() == 0 ){
32          bd = new BigDecimal(value.intValue());
33        }
34      }
35      return bd;
36    }

38  }
